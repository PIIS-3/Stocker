1
Crear el modelo y los schemas
models/store.py

Define las 4 clases: StoreBase (validaciones compartidas), StoreCreate (payload POST), StoreUpdate (payload PATCH, todo opcional) y Store (tabla ORM).

from typing import Optional, List, TYPE_CHECKING
from datetime import datetime
from sqlmodel import SQLModel, Field, Relationship
from .enums import StatusEnum
from .mixins import TimestampMixin

if TYPE_CHECKING:
    from .employee import Employee

class StoreBase(SQLModel):
    store_name: str = Field(min_length=1, description="Nombre único de la tienda.")
    address:    str = Field(min_length=1, description="Dirección física.")
    status: StatusEnum = Field(default=StatusEnum.Active, description="Estado operativo.")

class StoreCreate(StoreBase):
    pass

class StoreUpdate(SQLModel):
    store_name: Optional[str]        = Field(default=None, min_length=1)
    address:    Optional[str]        = Field(default=None, min_length=1)
    status:     Optional[StatusEnum] = None

class Store(StoreBase, TimestampMixin, table=True):
    id_store: Optional[int] = Field(default=None, primary_key=True)
    employees: List["Employee"] = Relationship(back_populates="store")

class StoreResponse(StoreBase):
    id_store:   int
    created_at: Optional[datetime] = None
    updated_at: Optional[datetime] = None

Crear las funciones CRUD
crud/stores.py

6 funciones estándar. Siempre devolver None cuando no se encuentra el recurso — el endpoint decide el código HTTP.

from sqlmodel import Session, select
from .. import models

def get_stores(db: Session, skip: int = 0, limit: int = 100) -> list[models.Store]:
    """Lista paginada ordenada por id_store."""
    return db.exec(select(models.Store).offset(skip).limit(limit)).all()

def get_store_by_id(db: Session, store_id: int) -> models.Store | None:
    return db.get(models.Store, store_id)

def get_store_by_name(db: Session, store_name: str) -> models.Store | None:
    return db.exec(
        select(models.Store).where(models.Store.store_name == store_name)
    ).first()

def create_store(db: Session, store_in: models.StoreCreate) -> models.Store:
    db_store = models.Store.model_validate(store_in)
    db.add(db_store)
    db.commit()
    db.refresh(db_store)  # Necesario para recoger id y timestamps de PostgreSQL
    return db_store

def update_store(db: Session, store_id: int, store_in: models.StoreUpdate) -> models.Store | None:
    db_store = db.get(models.Store, store_id)
    if not db_store:
        return None
    # exclude_unset=True: solo actualiza los campos que el cliente envió explícitamente
    data = store_in.model_dump(exclude_unset=True)
    for key, val in data.items():
        setattr(db_store, key, val)
    db.add(db_store)
    db.commit()
    db.refresh(db_store)
    return db_store

def delete_store(db: Session, store_id: int) -> models.Store | None:
    db_store = db.get(models.Store, store_id)
    if not db_store:
        return None
    db.delete(db_store)
    db.commit()
    return db_store  # Devolvemos el objeto antes de borrarlo para la respuesta
4
Crear los endpoints
api/endpoints/stores.py

Los GET usan get_current_employee (cualquier empleado autenticado). Los POST, PATCH y DELETE usan get_current_admin (solo SuperAdmin o Manager).

from fastapi import APIRouter, Depends, HTTPException, status
from sqlmodel import Session
from typing import List
from ... import models
from ...crud import stores as crud_stores
from ...database import get_db
from ..deps import get_current_employee, get_current_admin

# Router con auth mínima (cualquier empleado) para todos los endpoints
router = APIRouter(tags=["Tiendas"], dependencies=[Depends(get_current_employee)])

# Respuestas documentadas en Swagger
_404 = {404: {"description": "Tienda no encontrada."}}
_409 = {409: {"description": "Ya existe una tienda con ese nombre."}}

@router.get("/", response_model=List[models.StoreResponse], summary="Listar tiendas")
def read_stores(skip: int = 0, limit: int = 100, db: Session = Depends(get_db)):
    return crud_stores.get_stores(db, skip=skip, limit=limit)

@router.post(
    "/",
    response_model=models.StoreResponse,
    status_code=status.HTTP_201_CREATED,
    responses=_409,
    summary="Registrar nueva tienda",
    dependencies=[Depends(get_current_admin)],  # Solo admins pueden crear
)
def create_store(store_in: models.StoreCreate, db: Session = Depends(get_db)):
    if crud_stores.get_store_by_name(db, store_name=store_in.store_name):
        raise HTTPException(status_code=409, detail="Ya existe una tienda con ese nombre.")
    return crud_stores.create_store(db, store_in=store_in)

@router.get("/{store_id}", response_model=models.StoreResponse, responses=_404)
def read_store(store_id: int, db: Session = Depends(get_db)):
    db_store = crud_stores.get_store_by_id(db, store_id=store_id)
    if not db_store:
        raise HTTPException(status_code=404, detail="Tienda no encontrada.")
    return db_store

@router.patch("/{store_id}", response_model=models.StoreResponse,
              responses={**_404, **_409}, dependencies=[Depends(get_current_admin)])
def update_store(store_id: int, store_in: models.StoreUpdate, db: Session = Depends(get_db)):
    if store_in.store_name:
        existing = crud_stores.get_store_by_name(db, store_name=store_in.store_name)
        if existing and existing.id_store != store_id:
            raise HTTPException(status_code=409, detail="Ya existe una tienda con ese nombre.")
    updated = crud_stores.update_store(db, store_id=store_id, store_in=store_in)
    if not updated:
        raise HTTPException(status_code=404, detail="Tienda no encontrada.")
    return updated

@router.delete("/{store_id}", response_model=models.StoreResponse,
              responses=_404, dependencies=[Depends(get_current_admin)])
def delete_store(store_id: int, db: Session = Depends(get_db)):
    deleted = crud_stores.delete_store(db, store_id=store_id)
    if not deleted:
        raise HTTPException(status_code=404, detail="Tienda no encontrada.")
    return deleted
